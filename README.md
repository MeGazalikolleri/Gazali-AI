      };

      // Check for exact matches first
      let matched = false;
      for (const key in knowledge) {
        if (s === key.toLowerCase()) {
          answer = knowledge[key];
          matched = true;
          break;
        }
      }
      
      // If no exact match, check for partial matches

        for (const key in knowledge) {
